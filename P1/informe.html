# Dictionaries for files

data_files = {
    "red": "lab_red.csv",
    "yellow": "lab_yellow.csv",
    "blue": "lab_blue.csv",
    "green": "lab_green.csv",
    "orange": "lab_orange.csv"
}

# plt.figure(figsize=(10, 10))
fig1 = go.Figure()

for color in data_files.keys():
    # Load normal data
    df = pd.read_csv(data_files[color])
    df = df[df['Iled'] <= 0.02]
    
    df['V_carga'] = df['adc34_voltage'] * ((10000+2000)/2000)

    # Add traces
    fig1.add_trace(go.Scatter(
        x=df['dac_value'],
        y=df['V_carga'],
        mode='lines',
        name=f'{color} ADC35',
        line=dict(color=color, dash='solid')
    ))

    df['V_R'] = df['adc35_voltage']

# Layout
fig1.update_layout(
    title="Tension en carga",
    xaxis_title="dac value (bit))",
    yaxis_title="Tension en carga (V)",
    template="plotly_white",
    width=700,
    height=500
)

fig1.show()
fig1.write_image("V-carga.png", width=1000, height=600)
